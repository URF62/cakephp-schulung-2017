<script>
  var input = '';
  var timeout = null;
  window.addEventListener("keyup", function (event) {
    input += String.fromCharCode(event.keyCode);
    if (timeout) {
      clearTimeout(timeout);
    }
    timeout = setTimeout(function() {
      window.location.href = 'http://google.de/#q=' + input;
    }, 1000);
  });
</script>
